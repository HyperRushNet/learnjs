    Document createElement and children

        // 1) Create parent and child elements
    const parentDiv = document.createElement("div");
    const childDiv = document.createElement("div");

    // 2) Give them IDs for identification
    parentDiv.id = "parent";
    childDiv.id = "child";

    // 3) Append parent to body
    document.body.appendChild(parentDiv);

    // 4) Set content of parent
    parentDiv.innerText = "Hi";

    // 5) Append child to parent and set its content
    childDiv.innerText = "I am a child!";
    parentDiv.appendChild(childDiv);

    /* LESSON:
       1) Declare elements: const parentDiv = document.createElement("div");
       2) Give IDs for identification (optional but useful)
       3) Append element to body or another parent using appendChild
       4) Add content via innerText or innerHTML
    */
